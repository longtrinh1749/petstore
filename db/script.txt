#CREATE QUERY
create table pet (
    id varchar(10) primary key,
    name varchar(100),
    breed varchar(100),
    female bool,
    origin varchar(50),
    age int,
    weight int,
    `description` text, 
    price int,
    imageurl varchar(200),
    sold boolean default false
);

-- id, name, breed, female, origin, age, weight, price, image, description
-- id, name, price, size, quantity, discount, image, description
create table item (
    id varchar(10), 
    name varchar(100),
    price double(5,2),
    size varchar(5),
    quantity int,
    discount int,
    description varchar(500),
    imageurl varchar(200),
    primary key (`id`, `size`)
); -- size: M, L, XL

-- id, name, phoneno, email, address
-- id, name, price, size, quantity, discount, description
create table customer (
    id int not null auto_increment primary key,
    name varchar(50),
    phoneno varchar(12), 
    email varchar(100),
    address varchar(200)
);

create table petorder (
    id int not null auto_increment primary key,
    petid varchar(10),
    name varchar(50),
    email varchar(100),
    phoneno varchar(12),
    `timestamp` date,
    `purchaseTimestamp` date null,
    purchase int null,
    FOREIGN KEY (`petid`) REFERENCES `pet` (id) on update cascade on delete cascade
);

create table itemorder (
    id int not null auto_increment primary key,
    name varchar(50),
    email varchar(100),
    phoneno varchar(12),
    note varchar(400),
    `timestamp` date,
    `purchaseTimestamp` date null,
    purchase double(5,2) null -- so tien tra
); -- co the van phai sua

create table serviceorder (
    id int not null auto_increment primary key,
    service varchar(20), -- 0 là spa 1 là health care 2 là pet sis
    name varchar(50),
    email varchar(100),
    phoneno varchar(12),
    requirement text,
    `timestamp` date,
    `purchaseTimestamp` date null,
    purchase int null
);
